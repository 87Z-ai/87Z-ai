<script>
    const REPO_OWNER = "87z-ai";
    const REPO_NAME = "87z-ai";
    let ADMIN_TOKEN = "", CURRENT_ROOM = "", IS_ADMIN = false, currentSha = null;

    async function initializeWall() {
        const roomVal = document.getElementById('room-input').value;
        ADMIN_TOKEN = document.getElementById('token-input').value;
        CURRENT_ROOM = `raum_${roomVal}.json`;
        if(ADMIN_TOKEN) IS_ADMIN = true;
        document.getElementById('login-interface').style.display = 'none';
        document.getElementById('chat-interface').style.display = 'block';
        if(IS_ADMIN) document.getElementById('admin-panel').style.display = 'block';
        loadChat();
    }

    async function loadChat() {
        try {
            const res = await fetch(`https://api.github.com/repos/${REPO_OWNER}/${REPO_NAME}/contents/${CURRENT_ROOM}?nocache=${Date.now()}`);
            if(res.ok) {
                const data = await res.json();
                currentSha = data.sha;
                const messages = JSON.parse(atob(data.content));
                document.getElementById('chat-flow').innerHTML = messages.map(m => `<div>${m}</div>`).join('');
            }
        } catch(e) {}
    }

    async function sendMessage() {
        const text = document.getElementById('message-input').value;
        let msgs = [];
        try {
            const res = await fetch(`https://api.github.com/repos/${REPO_OWNER}/${REPO_NAME}/contents/${CURRENT_ROOM}`);
            if(res.ok) msgs = JSON.parse(atob((await res.json()).content));
        } catch(e) {}
        msgs.push(text);
        const headers = ADMIN_TOKEN ? {'Authorization': `token ${ADMIN_TOKEN}`} : {};
        await fetch(`https://api.github.com/repos/${REPO_OWNER}/${REPO_NAME}/contents/${CURRENT_ROOM}`, {
            method: 'PUT',
            headers: {...headers, 'Content-Type': 'application/json'},
            body: JSON.stringify({message: "Chat", content: btoa(JSON.stringify(msgs)), sha: currentSha})
        });
        loadChat();
    }

</script>
